#LEVEL-2
#Find ther total amount spent by each customer
SELECT
c.customerName,
SUM(od.Quantity * p.price) AS Total_Amount
FROM customers AS c
INNER JOIN orders AS o
ON c.customerID = o.customerID
INNER JOIN orderDetails AS od
ON o.orderID = od.orderID
INNER JOIN products AS p
ON od.productID = p.productID
GROUP BY c.customerID, c.customerName;

#display customers who have placed more than 2 orders
SELECT
c.customerName,
COUNT(o.orderID) AS Total_Orders
FROM customers AS c
INNER JOIN orders AS o
ON c.customerID = o.customerID
GROUP BY c.customerID, c.customerName
HAVING COUNT(o.orderID) > 2;

#Find the avg price of products supplied by each supplier
SELECT
s.supplierName,
AVG(p.price) AS Average_Price
FROM supplier AS s
INNER JOIN products AS p
ON s.supplierID = p.supplierID
GROUP BY s.supplierID, s.supplierName;

#find customers who purchased products from more than 1 supplier
SELECT
c.customerName,
COUNT(DISTINCT p.supplierID) AS Total_Suppliers
FROM customers AS c
INNER JOIN orders AS o
ON c.customerID = o.customerID
INNER JOIN orderDetails AS od
ON o.orderID = od.orderID
INNER JOIN products AS p
ON od.productID = p.productID
GROUP BY c.customerID, c.customerName
HAVING COUNT(DISTINCT p.supplierID) > 1;
