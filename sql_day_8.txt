/* STORED PROCEDURES
 normally SQL statements end with;
Inside a procedure there can be multiple SQL statements
So we temporary change the delimeter
*/

#QUERY TO CREATE PROCEDURE
DELIMITER //       #IT is not mandatory to use // we can use any symbol like $ 
CREATE PROCEDURE ShowAllEmployee()
BEGIN
	SELECT * FROM employees;
END //
DELIMITER ;

#EXECUTE
CALL ShowAllEmployee();


#DEPT FILTER

DELIMITER //
CREATE PROCEDURE EmployeeDepartment
(
IN dept VARCHAR(30)
)
BEGIN 
	SELECT * FROM employees
    WHERE department=dept;
END //
DELIMITER //
CALL EmployeeDepartment('IT');
CALL EmployeeDepartment('HR');
CALL EmployeeDepartment('Finance');

#SALARY FILTER
DROP PROCEDURE SalaryGreaterThan;
DELIMITER //
CREATE PROCEDURE SalaryGreaterThan
(
INamount DECIMAL(10,2)
)
BEGIN
    SELECT * FROM employees WHERE salary > amount;
END //
CALL SalaryGreaterThan(50000);

DELIMITER //
CREATE PROCEDURE EmployeeFilter
(
IN dept VARCHAR(30),
in MINsALARY decimal(10,2)
)
BEGIN
	SELECT* FROM employees WHERE department = dept 
    AND salary >= minSalary;
END //
DELIMITER ;
CALL EmployeeFilter('IT',55000);

SHOW PROCEDURE STATUS;
